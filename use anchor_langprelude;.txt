use anchor_lang::prelude::*;
use anchor_lang::solana_program::program::invoke;
use anchor_lang::solana_program::system_instruction;

declare_id!("3pitCpDAFY5jtyYvTUKrUuMHT19GqzA2sea8jhvjJBRq");

// Commission Address Constant
pub const COMMISSION_ADDRESS: Pubkey = solana_program::pubkey!("3FSdF5cDCjkEsrcLEeCDkNBkLLHtpbSiqpxTbcENnydJ");


#[program]
pub mod solana_coinflip {
    use super::*;

    pub fn create_game(
        ctx: Context<CreateGame>,
        game_id: u64,
        amount: u64,
        side: u8,
    ) -> Result<()> {
        let game = &mut ctx.accounts.game;
        game.player_one = ctx.accounts.player_one.key();
        game.player_two = Pubkey::default();
        game.amount = amount;
        game.player_one_side = side;
        game.game_id = game_id;
        game.status = 0; // 0: Open

        let ix = system_instruction::transfer(&ctx.accounts.player_one.key(), &game.key(), amount);
        invoke(
            &ix,
            &[
                ctx.accounts.player_one.to_account_info(),
                game.to_account_info(),
                ctx.accounts.system_program.to_account_info(),
            ],
        )?;

        emit!(GameCreated {
            game_id,
            player_one: game.player_one,
            amount
        });
        Ok(())
    }

    pub fn join_and_flip(ctx: Context<JoinAndFlip>) -> Result<()> {
        let game = &mut ctx.accounts.game;
        require!(game.status == 0, CoinflipError::GameAlreadyFinished);

        // 1. Set Player Two and collect their bet
        game.player_two = ctx.accounts.player_two.key();

        let ix = system_instruction::transfer(&ctx.accounts.player_two.key(), &game.key(), game.amount);
        invoke(
            &ix,
            &[
                ctx.accounts.player_two.to_account_info(),
                game.to_account_info(),
                ctx.accounts.system_program.to_account_info(),
            ],
        )?;

        // 2. Perform the Flip immediately
        let clock = Clock::get()?;
        let flip_result = (clock.unix_timestamp % 2) as u8;

        let winner_pubkey = if flip_result == game.player_one_side {
            game.player_one
        } else {
            game.player_two
        };

        // 3. Automated Payout with 0.1% Commission
        game.status = 2; // Finished
        let total_pool = game.amount * 2;
        
        // Calculate 0.1% (total_pool / 1000)
        let commission_amount = total_pool / 1000;
        let winner_prize = total_pool - commission_amount;

        let winner_info = if winner_pubkey == game.player_one {
            ctx.accounts.player_one.to_account_info()
        } else {
            ctx.accounts.player_two.to_account_info()
        };

        // Deduct total from game account
        **game.to_account_info().try_borrow_mut_lamports()? -= total_pool;
        
        // Transfer to winner
        **winner_info.try_borrow_mut_lamports()? += winner_prize;
        
        // Transfer to commission account
        **ctx.accounts.commission_account.try_borrow_mut_lamports()? += commission_amount;

        // 4. Emit the result
        emit!(FlipFinished {
            game_id: game.game_id,
            winner: winner_pubkey,
            player_one: game.player_one,
            player_two: game.player_two,
            amount: winner_prize, // Log the actual amount won
            side_won: flip_result,
        });

        Ok(())
    }
}

#[derive(Accounts)]
#[instruction(game_id: u64, amount: u64)]
pub struct CreateGame<'info> {
    #[account(
        init, payer = player_one,
        space = 8 + 32 + 32 + 8 + 1 + 8 + 1,
        seeds = [b"game", player_one.key().as_ref(), &game_id.to_le_bytes()],
        bump
    )]
    pub game: Account<'info, Game>,
    #[account(mut)]
    pub player_one: Signer<'info>,
    pub system_program: Program<'info, System>,
}

#[derive(Accounts)]
pub struct JoinAndFlip<'info> {
    #[account(
        mut, 
        seeds = [b"game", player_one.key().as_ref(), &game.game_id.to_le_bytes()], 
        bump
    )]
    pub game: Account<'info, Game>,
    #[account(mut)]
    pub player_one: AccountInfo<'info>,
    #[account(mut)]
    pub player_two: Signer<'info>,
    /// CHECK: This is the hardcoded commission address
    #[account(mut, address = COMMISSION_ADDRESS)]
    pub commission_account: AccountInfo<'info>,
    pub system_program: Program<'info, System>,
}

#[account]
pub struct Game {
    pub player_one: Pubkey,
    pub player_two: Pubkey,
    pub amount: u64,
    pub player_one_side: u8,
    pub game_id: u64,
    pub status: u8,
}

#[event]
pub struct GameCreated {
    pub game_id: u64,
    pub player_one: Pubkey,
    pub amount: u64,
}

#[event]
pub struct FlipFinished {
    pub game_id: u64,
    pub winner: Pubkey,
    pub player_one: Pubkey,
    pub player_two: Pubkey,
    pub amount: u64,
    pub side_won: u8,
}

#[error_code]
pub enum CoinflipError {
    #[msg("Already finished")]
    GameAlreadyFinished,
}
